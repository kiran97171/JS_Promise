Que 4 : "Create a promise function accepting an argument, if yes is passed to the function then it 

         should go to resolved state and print Promise Resolved, and if nothing is passed 
        then it should go to reject the state and catch the error and print Promise Rejected "

Ans:

function print(a){
      return new Promise((res,rej)=>{
        if(a==="yes"){
          res("Promise Resolved");
        }
        else{
          rej("Promise Rejected");
        }
      });
    }
    print("yes")
    .then((result)=>{console.log(result)})
    .catch((error)=>{console.log(error)})

Que 7 : Create examples to explain promises function

Ans:

function greet(a) {
      return new Promise((resolve, reject) => {
          if (a==="Good Morning") {
            resolve("Good Morning!");
          } else {
            reject("Good Night!");
          }
    })
    }
    greet("Good Morning")
      .then((result) => {console.log(result)})
      .catch((error) => {console.log(error)});
